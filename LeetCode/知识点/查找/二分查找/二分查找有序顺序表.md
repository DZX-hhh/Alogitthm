#### code

```c++
//二分查找
int Binare_Search(SSTable ST, ElemType value) {
	int low = 0, high = ST.TabLen - 1;
	while (low <= high) {
		int mid = (low + high) / 2;//中间位置
		if (ST.arr[mid] == value) {//查找成功
			return mid;
		} else if (ST.arr[mid] > value) {
			high = mid - 1;
		} else {
			low = mid + 1;
		}
	}
	return -1;
}
```
